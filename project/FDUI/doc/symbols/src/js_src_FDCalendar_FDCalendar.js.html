  6 var FDCalendar = function(options) {
  7 	FDLib.implement(this,FDControl);
  8 	this.options = FDLib.util.apply(this.getOptions(),options);
  9 	this.dateUtil = FDLib.date;
 10 	this.today = new Date();
 11 	// 年月日
 12 	this.date = new Date();	
 13 	
 14 	this.calendarDiv = document.createElement(FDTag.DIV);
 15 	this.calendarDiv.style.display = "none";
 16 	
 17 	this.table = this._buildCalendarTable();
 18 	this.thead = this.table.createTHead();
 19 	this.tbody = document.createElement(FDTag.TBODY);
 20 	this.tfoot = this.table.createTFoot();
 21 	this.table.appendChild(this.tbody);
 22 	
 23 	this.calendarDiv.appendChild(this.table);
 24 	
 25 	
 26 	this.selectedTD;
 27 	this.hourSelector;
 28 	this.minuteSelector;
 29 	
 30 	this._buildCalendarBody();
 31 	
 32 	this.setStyle('fd-calendar');
 33 	
 34 	this.renderToDom(document.body);
 35 }

384 	,_buildCalendar:function() {
385 		var self = this;
386 		// 当月天数
387 		var daysInMonth = this.dateUtil.getEndDate(this.getYear(),this.getMonth());
388 		// 每月1号
389 		var firstDayOfMonth = new Date(this.getYear(),this.getMonth() - 1,1);
390 		// 每月1号星期几
391 		var firstDay = firstDayOfMonth.getUTCDay();
392 		// 今天
393 		var today = new Date();
394 		// 日期
395 		var dayIndex = 1;
396 		// 日期行的索引
397 		var rowIndex = 0;
398 		
399 		var thirdRow = this.tbody.insertRow(rowIndex++);
400 		
401 		// 填充空白天数
402 		for(var i=0;i<=firstDay;i++){
403 			thirdRow.insertCell(i);
404 		}
405 		// 填充第一个星期
406 		for (var i=firstDay + 1;i<=6;i++){
407 			this._buildTD(thirdRow,i,dayIndex++);
408 		}
409 		// 填充剩下的天数
410 		while (dayIndex <= daysInMonth) {
411 			
412 			var row = this.tbody.insertRow(rowIndex++)
413 			for (var i=0;i<=6 && dayIndex <= daysInMonth;i++){
414 				this._buildTD(row,i,dayIndex++);
415 			}
416 		}
417 		
418 		this._initYearMonthText();
419 	}
420 	,_buildTD:function(row,cellIndex,date) {
421 		var td = row.insertCell(cellIndex);
422 		td.innerHTML = this.options.dayRender(td,date);
423 		this._setTDData(td,date);
424 		// 如果是今天
425 		if(this._isReachToToday(date)) {
426 			td.className = "today";
427 			td.title = "今天";
428 		}
429 		if(this._isSelectedDay(td)) {
430 			this._changeTdBgColor(td);
431 		}
432 		
433 	}
