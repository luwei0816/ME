1 /**
  2  * 文本框控件FDTextBox,继承自<a href="FDFieldComponent.html">FDFieldComponent</a><br>
  3  * @example 示例:<br>
  4  * <pre><code>
  5 var txtUsername;
  6 
  7 window.onload = function() {
  8 	txtUsername = new FDTextBox({name:"username",defaultValue:'请输入姓名'
  9 		,validates:[new FDValidate({errorMsg:'请输入正整数',items:['positiveInt']})]});
 10 	txtUsername.renderTo('username');
 11 	
 12 	var text2 = new FDTextBox({domId:'addr',width:200,height:30,name:'addr',label:'地址:',labelValign:'top'});
 13 	text2.addEvent('change',function(){
 14 		var e = FDLib.event.getEvent();
 15 		alert(e.target.value);
 16 	})
 17 }</code></pre>
 18 * @param options 参见<a href="FDFieldComponent.html#getOptions">FDFieldComponent.getOptions()</a><br>
 19 * @constructor
 20  */
 21 var FDTextBox = function(options) {
 22 	FDTextBox.superclass.constructor.call(this,options);
 23 	this.options = FDLib.util.apply(this.getOptions(),options);
 24 }
 25 
 26 FDLib.extend(FDTextBox,FDFieldComponent);
 27 
 28 /**
 29  * @private
 30  */
 31 //@override
 32 FDTextBox.prototype.getControlHtml = function() {
 33 	return '<input type="text" name="'+this.options.name+'" />';
 34 }
