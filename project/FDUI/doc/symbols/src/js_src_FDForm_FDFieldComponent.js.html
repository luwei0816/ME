1 /**
  2  * 具有值域的控件,继承自<a href="FDComponent.html">FDComponent</a><br>
  3  * @param options 参见{@link #getOptions}返回的对象
  4  * @constructor
  5  */
  6 var FDFieldComponent = function(optoins) {
  7 	FDFieldComponent.superclass.constructor.call(this,optoins);
  8 	FDLib.implement(this,FDField);
  9 	// 确保name完整性
 10 	this.options.name = this.options.name || "name_" + FDControl.generateCount();
 11 	// 存放form控件的div,如input,textarea
 12 	this.controlDiv = document.createElement(FDTag.DIV);
 13 	
 14 	this.msgId = this.options.name + "_msg";
 15 	this.msgDom;
 16 	
 17 	this._buildFormElHtml();
 18 	this._buildContent();
 19 	
 20 	this.setBounds(this.options.width,this.options.height);
 21 	
 22 	this.setStyle('fdform');
 23 	
 24 	this.reset();
 25 	
 26 	this.show();
 27 }

 34 /**
 35  * 覆盖父类方法,获取组件的默认属性
 36    @return 返回json数据类型<pre><code>
 37 {
 38 	domId:null // 定位的节点ID
 39 	,label:"" // 控件前面要显示的文字
 40 	,labelAlign:'right' // label水平对齐方式
 41 	,labelValign:'middle' // label垂直对齐方式
 42 	,name:"" // 控件的name
 43 	,width:"" // 控件的宽,如'120px'
 44 	,height:"" // 控件的高,如'30px'
 45 	,validates:[] // 验证类
 46 	,defaultValue:'' // 默认值
 47 }</code></pre>
 48  */
 49 //@override
 50 FDFieldComponent.prototype.getOptions = function() {
 51 	var options = FDFieldComponent.superclass.getOptions.call(this);
 52 	return FDLib.util.apply(options,{
 53 		// 默认值
 54 		defaultValue:''
 55 	});
 56 }

243 /**
244  * 构建控件内容
245  * 格式:
246  * ┏━━━━━━━━━━━━━━┓
247  * ┃姓名:┃ 控件部分 ┃提示信息 ┃
248  * ┗━━━━━━━━━━━━━━┛
249  * @private
250  */
251 FDFieldComponent.prototype._buildContent = function() {
252 	var table = document.createElement(FDTag.TABLE);
253 	var tbody = document.createElement(FDTag.TBODY);
254 	table.appendChild(tbody);
255 	
256 	var tr = tbody.insertRow(0);
257 	var cellIndex = 0;
258 	
259 	if(this.options.label && FDLib.util.isString(this.options.label)) {
260 		this._buildLabelTD(tr,cellIndex++);
261 	}
262 	this._buildFormElTD(tr,cellIndex++);
263 	this._buildMsgTD(tr,cellIndex++);
264 	
265 	this.outerDiv.appendChild(table);
266 }
267 
268 FDFieldComponent.prototype._buildLabelTD = function(tr,cellIndex) {
269 	var tdLabel = tr.insertCell(cellIndex);
270 	tdLabel.className = 'lab';
271 	tdLabel.align = this.options.labelAlign;
272 	tdLabel.vAlign = this.options.labelValign;
273 	
274 	tdLabel.innerHTML = this.options.label;
275 }
276 
277 FDFieldComponent.prototype._buildFormElTD = function(tr,cellIndex) {
278 	var tdFormEl = tr.insertCell(cellIndex);
279 	tdFormEl.appendChild(this.controlDiv);
280 }

288 }
289 
290 FDFieldComponent.prototype._buildMsgTD = function(tr,cellIndex) {
291 	var tdMsg = tr.insertCell(cellIndex);
292 	this.msgDom = document.createElement(FDTag.DIV);
293 	this.msgDom.className = "msg";
294 	this.msgDom.id = this.msgId;
295 	tdMsg.appendChild(this.msgDom);
296 }

298 /**
299  * 验证
300  */
301 FDFieldComponent.prototype.validate = function() {
302 	var value = this.getValue();
303 	var validates = this.options.validates || [];
304 	
305 	for(var i=0,len=validates.length; i<len; i++) {
306 		var validate = validates[i];
307 		validate.setMsgId(this.msgId);
308 		var val = validate.validate(value);
309 		if(!val) {
310 			return false;
311 		}
312 	}
313 	return true;
314 }
