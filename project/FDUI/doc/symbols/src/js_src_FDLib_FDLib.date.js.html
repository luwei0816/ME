1 /**
  2  * 日期工具类
  3  * @class
  4  */
  5 FDLib.date = {
  6 	/**
  7 	 * 判断是否是闰年
  8 	 * @param year 年份
  9 	 * @return 闰年为true
 10 	 */
 11 	isLeapYear:function(year) {
 12 		return !!((year & 3) == 0 && (year % 100 || (year % 400 == 0 && year)));
 13 	}
 14 	/**
 15 	 * 是否是日期类型
 16 	 * @param dateStr 字符串日期
 17 	 * @return 是,true
 18 	 */
 19 	,isDateStr:function(dateStr) {
 20 		return !!this._parseDateStrToNum(dateStr);
 21 	}
 22 	/**
 23 	 * 根据年份获取每月的天数
 24 	 * @param year 年份
 25 	 * @return 返回每个月的天数,数组形式
 26 	 */
 27 	,getMonthDaysArr:function(year) {
 28 		// 存放每月天数
 29 		var month_days_arr = [31,28,31,30,31,30,31,31,30,31,30,31];
 30 		// 二月份天数
 31 		month_days_arr[1] = this.isLeapYear(year) ? 29 : 28;
 32 		
 33 		return month_days_arr;
 34 	}
 35 	/**
 36 	 * 根据年月得到该月的最后一天
 37 	 * @param year 年份
 38 	 * @param month 月份
 39 	 * @return 返回该月最后一天
 40 	 */
 41 	,getEndDate:function(year,month) {
 42 		// 存放每月天数
 43 		var month_days_arr = this.getMonthDaysArr(year);
 44 		return month_days_arr[month - 1];
 45 	}
 46 	/**
 47 	 * 将字符串日期转换成Date类型
 48 	 * 	
 49 	 * @param dateStr 字符串日期,如:
 50 	 *  var s = "2012-9-3 09:41:30";
 51 	 *  var s2 = "2012-09-03 09:41:30";
 52 	 *  var s3 = "9/3/2012 09:41:30";
 53 	 *  var s4 = "2012-9-3";
 54 	 * @return Date类型日期
 55 	 */
 56 	,parse:function(dateStr) {
 57 		return new Date(this._parseDateStrToNum(dateStr)); 
 58 	}
 59 	,_parseDateStrToNum:function(dateStr) {
 60 		return Date.parse(dateStr.replace(/-/g,"/"));
 61 	}
 62 	/**
 63 	 * 格式化日期<br>
 64 	 * 使用方法:
 65 	 * <code>
 66 	 * var dateStr = FDLib.date.format(new Date(),'yyyy-MM-dd hh:mm:ss.S');
 67 	 * </code>
 68 	 * 
 69 	 * @param dateInstance Date实例
 70 	 * @param 格式化字符串,如"yyyy-MM-dd","yyyy-MM-dd hh:mm:ss.S"
 71 	 * 
 72 	 * @return 返回格式化后的字符串
 73 	 */
 74 	,format:function(dateInstance,pattern) {
 75 		var o = {   
 76 			"M+" : dateInstance.getMonth()+1,                 //月份    
 77 			"d+" : dateInstance.getDate(),                    //日    
 78 			"h+" : dateInstance.getHours(),                   //小时    
 79 			"m+" : dateInstance.getMinutes(),                 //分    
 80 			"s+" : dateInstance.getSeconds(),                 //秒    
 81 			"q+" : Math.floor((dateInstance.getMonth()+3)/3), //季度    
 82 			"S"  : dateInstance.getMilliseconds()             //毫秒    
 83 		};   
 84 		if(/(y+)/.test(pattern)) {
 85 			pattern = pattern.replace(RegExp.$1, (dateInstance.getFullYear()+"").substring(4 - RegExp.$1.length));   
 86 		}
 87 			
 88 		for(var k in o) {
 89 			if(new RegExp("("+ k +")").test(pattern)) {
 90 				pattern = pattern.replace(RegExp.$1, (RegExp.$1.length == 1) 
 91 					? (o[k]) 
 92 					: (("00"+ o[k]).substring((""+ o[k]).length))); 
 93 			}
 94 		}
 95 			
 96 		return pattern; 
 97 	}
 98 };
