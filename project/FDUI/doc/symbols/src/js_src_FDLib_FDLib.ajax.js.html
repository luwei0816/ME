1 /**
  2  * Ajax工具类
  3  * @class
  4  */
  5 FDLib.ajax = {
  6 	/**
  7 	 * 提交请求
  8 	 * @param options
  9 	 * { url:'',params:{},success:function(){} }
 10 	 */
 11 	request:function(options) {
 12 		var url = options.url;
 13 		var params = options.params || {};
 14 		var callback = options.success || function(){};
 15 		
 16 		var xhr = this.createXhrObject();
 17 		
 18 		xhr.onreadystatechange = function() {
 19 			if(xhr.readyState !== 4) {
 20 				return;
 21 			}
 22 			if(xhr.status === 200 || xhr.status === 0) {
 23 				callback(xhr.responseText)
 24 			} else {
 25 				throw new Error('请求失败:' + xhr.responseText);
 26 			}
 27 		};
 28 		
 29 		xhr.open('POST',url,false);
 30 		xhr.setRequestHeader("Content-Type","application/x-www-form-urlencoded");
 31 		xhr.send(this._formatParamsToArr(params) || null);
 32 	}
 33 	,_formatParamsToArr:function(params) {
 34 		var retArr = [];
 35 		for(var key in params) {
 36 			var val = params[key];
 37 			var para = key + "=" + val;
 38 			if(this._isArray(val)) {
 39 				para = this._formatArrParam(key,val);
 40 			}
 41 			if(para) {
 42 				retArr.push(para);
 43 			}
 44 		}
 45 		if(retArr.length == 0) {
 46 			return '';
 47 		}
 48 		return retArr.join('&');
 49 	}
 50 	// 将数组转换成键值格式
 51 	//  id:[1,2,3] --> id=1&id=2&id=3
 52 	,_formatArrParam:function(key,vals) {
 53 		if(vals.length == 0) {
 54 			return '';
 55 		}
 56 		var ret = [];
 57 		for(var i=0,len=vals.length; i<len; i++) {
 58 			ret.push(key + "=" + vals[i]);
 59 		}
 60 		return ret.join('&');
 61 	}
 62 	,_isArray:function(o) {
 63 		return Object.prototype.toString.apply(o)
 64 			=== '[object Array]';
 65 	}
 66 	/**
 67 	 * 创建XHR对象
 68 	 * @private
 69 	 */
 70 	,createXhrObject:function() {
 71 		var methods = [
 72 			function(){ return new XMLHttpRequest();}
 73 			,function(){ return new ActiveXObject('Msxml2.XMLHTTP');}
 74 			,function(){ return new ActiveXObject('Microsoft.XMLHTTP');}
 75 		];
 76 		
 77 		for(var i=0,len=methods.length; i<len; i++) {
 78 			try {
 79 				methods[i]();
 80 			} catch (e) {
 81 				continue;
 82 			}
 83 			
 84 			this.createXhrObject = methods[i];
 85 			return methods[i]();
 86 		}
 87 		
 88 		throw new Error("创建XHR对象失败");
 89 	}
 90 };
