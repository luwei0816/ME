1 /**
  2  * 字符串工具类
  3  * @class
  4  */
  5 FDLib.string = {
  6 	/**
  7 	 * 格式化字符串
  8 	 * @example 用法:
  9 	 * FDLib.string.formatStr('{y}年{m}月{d}日',{y:'2010',m:'09',d:'15'});
 10 	 * 返回:2010年09月15日
 11 	 * 
 12 	 * @param str 需要格式化的字符串
 13 	 * @param obj json数据
 14 	 * @return 返回格式化后的字符串
 15 	 */
 16 	format:function(str,obj){
 17 		for(var key in obj){
 18 			str = str.replace(new RegExp("\\{\\s*" + key + "\\s*\\}", "g"), obj[key]);
 19 		}
 20 		return str;
 21 	}
 22 	/**
 23 	 * 是否是一个空字符串. ''或""
 24 	 * @return 是,true
 25 	 */
 26 	,isEmptyStr:function(value) {
 27 		return (typeof value === 'string' && !value);
 28 	}
 29 };
