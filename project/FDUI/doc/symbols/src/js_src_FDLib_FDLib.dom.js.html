  5 FDLib.dom = {
  6 	/**
  7 	 * 包裹节点,相当于jquery的warp
  8 	 * @param target dom对象
  9 	 * @param html 可以是dom对象也可以是string
 10 	 */
 11 	wrap:function(target,html) {
 12 		var wrap = html;
 13 		if(FDLib.util.isString(html)) {
 14 			if(document.createRange) {
 15 				var range = document.createRange();
 16 	      		range.selectNodeContents(target);
 17 	      		wrap = range.createContextualFragment(html).firstChild;
 18 			} else {
 19 		    	wrap = document.createElement(html);
 20 		    }
 21 		}
 22 		target.parentNode.replaceChild(wrap,target);
 23 		wrap.appendChild(target);
 24 		
 25 		return wrap;
 26 	}
 27 	/**
 28 	 * 隐藏节点
 29 	 * @param dom DOM对象
 30 	 */
 31 	,hideDom:function(dom) {
 32 		dom.style.display = "none";
 33 	}
 34 	/**
 35 	 * 显示节点
 36 	 * @param dom DOM对象
 37 	 */
 38 	,showDom:function(dom) {
 39 		dom.style.display = "block";
 40 	}
 41 	/**
 42 	 * 添加元素的class属性
 43 	 * @param dom DOM对象,如input,div等
 44 	 * @param classNames class名,字符串格式,多个用空格隔开.如:"enable","enable selected"
 45 	 */
 46 	,addClass:function(dom,classNames) {
 47 		var classNameArr = (classNames || "").split(/\s+/);
 48 		for(var i=0,len=classNameArr.length; i<len; i++) {
 49 			var className = classNameArr[i];
 50 			if(dom.nodeType == 1 && !this.hasClass(dom,className)) {
 51 				dom.className += (dom.className ? " " : "") + className;
 52 			}
 53 		}
 54 	}
 55 	/**
 56 	 * 移除元素的class属性
 57 	 * @param dom DOM对象,如input,div等
 58 	 * @param className class名,字符串格式,多个用空格隔开.如:"enable","enable selected"
 59 	 */
 60 	,removeClass:function(dom,classNames) {
 61 		if(dom.nodeType == 1) {
 62 			var classNameArr = (classNames || "").split(/\s+/);
 63 			var domClasses = (dom.className).toString().split(/\s+/);
 64 			for(var i=0,len=classNameArr.length; i<len; i++) {
 65 				var className = classNameArr[i];
 66 				for(var j=0,len2=domClasses.length; j<len2; j++) {
 67 					if(domClasses[j] == className) {
 68 						domClasses[j] = '';
 69 					}
 70 				}
 71 			}
 72 			dom.className = domClasses.join(' ');
 73 		}
 74 	}
 75 	/**
 76 	 * 判断DOM是否含有className
 77 	 * @param dom DOM 对象
 78 	 * @param className 字符串格式
 79 	 */
 80 	,hasClass:function(dom,className) {
 81 		var classNameArr = (dom.className).toString().split(/\s+/);
 82 		for(var i=0,len=classNameArr.length; i<len; i++) {
 83 			if(classNameArr[i] == className) {
 84 				return true;
 85 			}
 86 		}
 87 		return false;
 88 	}
 89 	/**
 90 	 * 给DOM对象绑定style
 91 	 * @param dom DOM对象
 92 	 * @param styleJson 存放样式,如:{'backgroundColor':'red','border':'1px'};采用驼峰式写法
 93 	 */
 94 	,bindDomStyle:function(dom,styleJson) {
 95 		for(var styleName in styleJson) {
 96 			dom.style[styleName] = styleJson[styleName];
 97 		}
 98 	}
 99 	/**
100 	 * 获取元素距离页面左边的值,相当于jquery中的offset().left
101 	 * @return 返回int类型的值
102 	 */
103 	,getOffsetX:function(element) {
104 		var actualLeft = element.offsetLeft;
105 		var current = element.offsetParent;
106 	
107 		while (current !== null){
108 			actualLeft += current.offsetLeft;
109 			current = current.offsetParent;
110 		}
111 	
112 		if (document.compatMode == "BackCompat"){
113 			var elementScrollLeft = document.body.scrollLeft;
114 		} else {
115 			var elementScrollLeft = document.documentElement.scrollLeft; 
116 		}
117 	
118 		return actualLeft - elementScrollLeft;
119 	}
120 	/**
121 	 * 获取元素距离页面顶部的值,相当于jquery中的offset().top
122 	 * @return 返回int类型的值
123 	 */
124 	,getOffsetY:function(element) {

126 		var current = element.offsetParent;
127

137 　　　　}
138

140 	}
141 	/**
142 	 * 等同于jquery.offset()
143 	 * @return 返回 {letf:int,top:int}
144 	 */
145 	,getOffset:function(dom) {

147 	}
