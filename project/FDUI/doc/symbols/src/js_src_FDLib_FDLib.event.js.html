1 /**
  2  * 事件工具类,需要加载FDLib.browser.js
  3  * @class
  4  */
  5 FDLib.event = {
  6 	/**
  7 	 * 添加事件
  8 	 * @param oTarget: DOM对象
  9 	 * @param sEventType:事件类型
 10 	 * @param fn:函数名
 11 	 */
 12 	addEvent : function(oTarget,sEventType,fn){
 13 		if(oTarget.addEventListener){
 14 			this.addEvent = function(oTarget,sEventType,fn) {
 15 				oTarget.addEventListener(sEventType,fn,false);
 16 			}
 17 		}else if(oTarget.attachEvent){
 18 			this.addEvent = function(oTarget,sEventType,fn) {
 19 				oTarget.attachEvent("on" + sEventType,fn);
 20 			}
 21 		}else{
 22 			this.addEvent = function(oTarget,sEventType,fn) {
 23 				oTarget["on" + sEventType] = fn;
 24 			}
 25 		}
 26 		// 调用新函数
 27 		this.addEvent(oTarget,sEventType,fn);
 28 	}
 29 	/**
 30 	 * 对一个父节点下面所有的子节点添加相同的事件
 31 	 * @param options <pre><code>
 32 {
 33 	superDom:menu // 父节点DOM对象
 34 	,tagName:'A'  // 子节点的类型
 35 	,eventName:'click' // 事件名
 36 	,handler:function(target) {  // 执行方法,target是子节点的DOM对象
 37 		iframe.src = target.href;
 38 		mainTitle.innerHTML = target.innerHTML;
 39 	}
 40 }</code></pre>
 41 	 */
 42 	,addBatchEvent:function(options) {
 43 		var self = this;
 44 		var superDom = options.superDom
 45 			,tagName = options.tagName
 46 			,eventName = options.eventName
 47 			,fn = options.handler;
 48 		this.addEvent(superDom,eventName,function(){
 49 			var e = self.getEvent();
 50 			var target = e.target;
 51 			
 52 			if(target.tagName !== tagName) {
 53 				return;
 54 			}
 55 			
 56 			fn(target);
 57 			
 58 			e.preventDefault();
 59 			e.stopPropagation();
 60 		})
 61 	}
 62 	/**
 63 	 * 移除事件
 64 	 * @param oTarget: DOM对象
 65 	 * @param sEventType:事件类型
 66 	 * @param fn:函数名,如果DOM对象上有多个click事件的话,可以指定移除哪一个函数
 67 	 */
 68 	,removeEvent : function(oTarget,sEventType,fn){
 69 		if(oTarget.removeEventListener){
 70 			this.removeEvent = function(oTarget,sEventType,fn) {
 71 				oTarget.removeEventListener(sEventType,fn,false);
 72 			}
 73 		}else if(oTarget.detachEvent){
 74 			this.removeEvent = function(oTarget,sEventType,fn) {
 75 				oTarget.detachEvent("on"+sEventType,fn);
 76 			}
 77 		}else {
 78 			this.removeEvent = function(oTarget,sEventType,fn) {
 79 				oTarget["on"+sEventType] = null;
 80 			}
 81 		}
 82 		
 83 		this.removeEvent(oTarget,sEventType,fn);
 84 	}
 85 	/**
 86 	 * 格式化事件对象,做到IE与DOM的统一
 87 	 * @param oEvent:事件对象
 88 	 */
 89 	,formatEvent : function(oEvent){
 90 		if(FDLib.browser.isIE()){
 91 			oEvent.charCode = (oEvent.type == "keypress")?oEvent.charCode:0;
 92 			oEvent.eventPhase = 2;
 93 			oEvent.isChar = (oEvent.charCode > 0);
 94 			oEvent.pageX = oEvent.clientX + document.body.scrollLeft;
 95 			oEvent.pageY = oEvent.clientY + document.body.scrollTop;
 96 			// 阻止某个事件的默认行为
 97 			oEvent.preventDefault = function(){
 98 				this.returnValue = false;
 99 			}
100 			
101 			if(oEvent.type == "mouseout"){
102 				oEvent.relateTarget = oEvent.toElement;
103 			} else if(oEvent.type == "mouseover"){
104 				oEvent.relateTarget = oEvent.fromElement;
105 			}
106 			
107 			// 阻止冒泡
108 			oEvent.stopPropagation = function(){
109 				this.cancelBubble = true;
110 			}
111 			
112 			oEvent.target = oEvent.srcElement;
113 			oEvent.timestamp = (new Date()).getTime();
114 		}
115 		return oEvent;
116 	}
117 	/**
118 	 * 格式化事件对象
119 	 */
120 	,getEvent : function(){
121 		if(window.event){
122 			return this.formatEvent(window.event);
123 		}else {
124 			return this.getEvent.caller.arguments[0];
125 		}
126 	}
127 	/**
128 	 * 点击div之外任何地方，隐藏div 
129 	 * @param showDom 点击showDom显示targtDom,点击页面其它地方隐藏targetDom
130 	 * @param targetDom 目标元素
131 	 */
132 	,registClickOtherHide:function(showDom,targetDom) {
133 		var self = this;
134 		this.addEvent(document,'click',function(){
135 			var e = self.getEvent();
136 		    var elem = e.target
137 		    while (elem) {  
138 		        if (elem != document) {  
139 		            if (elem === targetDom || elem === showDom) {  
140 		                targetDom.style.display = "block";
141 		                return;  
142 		            }  
143 		            elem = elem.parentNode;  
144 		        } else {  
145 		            targetDom.style.display = "none";
146 		            return;  
147 		        }  
148 		    }  
149 			
150 		})
151 	}
152 };
