1 /**
  2  * form组件的父类,提供默认的属性,和方法
  3  * @param options 参见{@link #getOptions}返回的对象
  4  * @constructor
  5  */
  6 var FDComponent = function(options) {
  7 	FDLib.implement(this,FDControl);
  8 	
  9 	this.options = FDLib.util.apply(this.getOptions(),options);
 10 	
 11 	this.outerDiv = document.createElement(FDTag.DIV);
 12 	this.hide();
 13 	
 14 	if(FDLib.util.isString(this.options.domId)) {
 15 		this.renderTo(this.options.domId);
 16 	}
 17 }
 18 
 19 FDComponent.prototype = {
 20 /**
 21  * 获取组件的默认属性
 22    @return <pre><code>返回json数据类型
 23 {
 24 	domId:null // 定位的节点ID
 25 	,label:"" // 控件前面要显示的文字
 26 	,labelAlign:'right' // label水平对齐方式
 27 	,labelValign:'middle' // label垂直对齐方式
 28 	,name:"" // 控件的name
 29 	,width:"" // 控件的宽,如'120px'
 30 	,height:"" // 控件的高,如'30px'
 31 	,validates:[] // 验证类
 32 }</code></pre>
 33  */
 34 	getOptions:function() {
 35 		return {
 36 			domId:null
 37 			,label:""
 38 			,labelAlign:'right'
 39 			,labelValign:'middle'
 40 			,name:""
 41 			,width:""
 42 			,height:""
 43 			,validates:[]
 44 		};
 45 	}
 46 	/**
 47 	 * 显示控件
 48 	 */
 49 	,show:function() {
 50 		this.outerDiv.style.display = "inline-block";
 51 	}
 52 	/**
 53 	 * 隐藏控件
 54 	 */
 55 	,hide:function() {
 56 		this.outerDiv.style.display = "none";
 57 	}
 58 	/**
 59 	 * 设置控件样式
 60 	 * @param style className,字符串格式
 61 	 */
 62 	,setStyle:function(style) {
 63 		if(FDLib.util.isString(style)) {
 64 			this.outerDiv.className = style;
 65 		} 
 66 	}
 67 	/**
 68 	 * 设置控件样式,作用同setStyle
 69 	 * @param className className,字符串格式
 70 	 */
 71 	,setClass:function(className) {
 72 		this.setStyle(className);
 73 	}
 74 	/**
 75 	 * 添加控件样式.<br>
 76 	 * <b>注:此样式添加在控件的最外层的div上</b>
 77 	 * @param className className,字符串格式
 78 	 */
 79 	,addClass:function(className) {
 80 		FDLib.dom.addClass(this.outerDiv,className);
 81 	}
 82 	/**
 83 	 * 移除控件样式
 84 	 * @param className className,字符串格式
 85 	 */
 86 	,removeClass:function(className) {
 87 		FDLib.dom.removeClass(this.outerDiv,className);
 88 	}
 89 };
