1 /**
  2  * 文本框控件,继承自<a href="FDFieldComponent.html">FDFieldComponent</a><br>
  3  * @example 示例:
  4  * <pre><code>
  5 var remark;
  6 window.onload = function() {
  7 	remark = new FDTextArea({domId:'remark'
  8 		//,width:300,height:80
  9 		,name:'addr',label:'地址:',labelValign:'top'
 10 		,validates:[new FDValidate({minLength:4,maxLength:10})]});
 11 }
 12  * </code></pre>
 13  * @param options 参见{@link #getOptions}返回的对象
 14  * @constructor
 15  * 2012-7-31
 16  */
 17 var FDTextArea = function(options) {
 18 	FDTextArea.superclass.constructor.call(this,options);
 19 	this.options = FDLib.util.apply(this.getOptions(),options);
 20 }
 21 
 22 FDLib.extend(FDTextArea,FDFieldComponent);
 23 
 24 /**
 25  * 覆盖父类方法,获取组件的默认属性
 26    @return <pre><code>返回json数据类型
 27 {
 28 	domId:null // 定位的节点ID
 29 	,label:"" // 控件前面要显示的文字
 30 	,labelAlign:'right' // label水平对齐方式
 31 	,labelValign:'top' // label垂直对齐方式
 32 	,name:"" // 控件的name
 33 	,width:240 // 控件的宽,如'120px'
 34 	,height:80 // 控件的高,如'30px'
 35 	,validates:[] // 验证类
 36 	,defaultValue:'' // 默认值
 37 }</code></pre>
 38  */
 39 //@override
 40 FDTextArea.prototype.getOptions = function() {
 41 	var options = FDTextArea.superclass.getOptions.call(this);
 42 	return FDLib.util.apply(options,{
 43 		width:240
 44 		,height:80
 45 		,labelValign:'top'
 46 	});
 47 }
 48 
 49 /**
 50  * @private
 51  */
 52 //@override
 53 FDTextArea.prototype.getControlHtml = function() {
 54 	return '<textarea name="' + this.options.name + '" ></textarea>';
 55 }
 56 
 57 /**
 58  * @private
 59  */
 60 // 重写父类的方法
 61 //@override
 62 FDTextArea.prototype.getControlItems = function() {
 63 	return this.controlDiv.getElementsByTagName('textarea');
 64 }
