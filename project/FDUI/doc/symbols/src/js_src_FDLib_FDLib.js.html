1 /**
  2  * 提供类的接口创建,类的继承操作
  3  * @example 示例:
  4  * var Book = createInterface(["getBookName","getPunishDate"]);
  5  * FDLib.extend(subClass,supClass);
  6  * @class
  7  */
  8 var FDLib = (function(){
  9 	var doc = document;
 10 	
 11 	var Interface = function(_methods) {
 12 		if (arguments.length != 1) {
 13 			throw new Error("创建接口参数只能有一个,并且为数组类型");
 14 		}
 15 		this.methods = [];
 16 		this._isInterfaceType = true;
 17 
 18 		for (var i = 0, len = _methods.length; i < len; i++) {
 19 			if (typeof _methods[i] !== "string") {
 20 				throw new Error("定义接口中的方法必须为字符串类型");
 21 			}
 22 			this.methods.push(_methods[i]);
 23 		}
 24 	}
 25 	
 26 	return {
 27 		/**
 28 		 * 创建接口
 29 		 * 如:var Book = createInterface(["getBookName","getPunishDate"]);
 30 		 * @param:methods:接口中的方法数组类型 
 31 		 * 
 32 		 */
 33 		createInterface : function(methods) {
 34 			return new Interface(methods);
 35 		}
 36 		/**
 37 		 * 继承原型
 38 		 * 子类中调用父类的函数:
 39 		 * Man.superclass.sayName.call(this);
 40 		 * 子类中调用父类构造函数:
 41 		 * Man.superclass.constructor.call(this,param);
 42 		 * @param subClass 子类函数名
 43 		 * @param superClass 父类函数名
 44 		 */
 45 		,extend:function(subClass,superClass){
 46 			var F = function(){};
 47 			F.prototype = superClass.prototype;
 48 			subClass.prototype = new F();			
 49 			subClass.prototype.constructor = subClass;
 50 			
 51 			subClass.superclass = superClass.prototype;
 52 			if(superClass.prototype.constructor == Object.prototype.constructor) {
 53 				superClass.prototype.constructor = superClass;
 54 			}
 55 		}
 56 		/**
 57 		 * 实现接口
 58 		 * implement(subClassInstance,interface1,interface2,...);
 59 		 * @param subClassInstance 子类对象实例
 60 		 */
 61 		,implement:function(subClassInstance){			
 62 			for(var i=1,len = arguments.length; i<len; i++) {
 63 				var interfac = arguments[i];
 64 				if(!interfac._isInterfaceType) {
 65 					throw new Error("类必须实现接口类型");
 66 				}
 67 				for(var j=0,methodLen = interfac.methods.length; j<methodLen; j++) {
 68 					var method = interfac.methods[j];
 69 					if(!subClassInstance[method] 
 70 						|| typeof subClassInstance[method] !== "function") {
 71 						throw new Error("类没有实现接口中的[ " + method + " ]方法");
 72 					}
 73 				}
 74 			}
 75 		}
 76 		/**
 77 		 * 根据id获取dom节点
 78 		 * @return 返回dom对象
 79 		 */
 80 		,getEl:function(id) {
 81 			return doc.getElementById(id);
 82 		}
 83 	};
 84 })();
