1 /**
  2  * JString对象,功能类似于Java中的StringBuilder
  3  * @example 示例:
  4 var str = new JString();
  5 str.append('Hello').append(' World');
  6 alert(str.toString()); // Hello World
  7  * @class
  8  */
  9 function JString(s){
 10 	this._strings = new Array;
 11 	if(s && typeof s === 'string') {
 12 		this._strings.push(s);
 13 	}
 14 }
 15 
 16 /**
 17  * 追加字符串
 18  * @return 返回JString对象
 19  */
 20 JString.prototype.append = function(s){
 21 	this._strings.push(s);
 22 	return this;
 23 }
 24 
 25 /**
 26  * 返回字符串
 27  */
 28 JString.prototype.toString = function(){
 29 	return this._strings.join("");
 30 }
