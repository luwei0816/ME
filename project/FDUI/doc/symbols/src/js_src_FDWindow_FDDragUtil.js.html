1 /**
  2  * 窗体拖拽工具类
  3  * @example 示例:
  4  * // 注册
  5  * FDDragUtil.regist(winDom,titleDom);
  6  * // 取消拖拽
  7  * FDDragUtil.destory(titleDom);
  8  * @class
  9  */
 10 var FDDragUtil = (function(){
 11 
 12 	var doc = document;
 13 	
 14 	var moveX = 0;
 15 	var moveY = 0;
 16 	var moveTop = 0;
 17 	var moveLeft = 0;
 18 	var moveable = false;
 19 	
 20 	return {
 21 		/**
 22 		 * 注册拖拽
 23 		 * 需要传入整个窗体dom和标题部分的dom
 24 		 */
 25 		regist:function(winDom,titleDom) {	
 26 			var width = doc.body.clientWidth; 
 27 			var height = doc.body.clientHeight;
 28 			var title = titleDom;
 29 			var win = winDom;
 30 			var winStyle = win.style;
 31 			title.style.cursor = 'move';
 32 
 33 			title.onmousedown = function() {
 34 				var evt = FDDragUtil._getEvent();	
 35 				
 36 				moveable = true; 
 37 				moveX = evt.clientX;
 38 				moveY = evt.clientY;
 39 				
 40 				moveTop = parseInt(winStyle.top);
 41 				moveLeft = parseInt(winStyle.left);
 42 				
 43 				doc.onmousemove = function() {
 44 					if (moveable) {
 45 						var evt = FDDragUtil._getEvent();
 46 						var x = moveLeft + evt.clientX - moveX;							
 47 						var y = moveTop + evt.clientY - moveY;
 48 						if ( x > 0 && y > 0) {
 49 							winStyle.left = x + "px";
 50 							winStyle.top = y + "px";
 51 						}
 52 					}	
 53 				};
 54 				doc.onmouseup = function () { 
 55 					if (moveable) { 					
 56 						moveable = false; 
 57 						moveX = 0;
 58 						moveY = 0;
 59 						moveTop = 0;
 60 						moveLeft = 0;
 61 					} 
 62 				};
 63 			}
 64 		}
 65 		,moveToCenter:function(win) {
 66 			var body = doc.body
 67 				,docEl = doc.documentElement;
 68 				
 69 			var clientHeight = docEl.clientHeight || body.clientHeight
 70 				,clientWidth = docEl.clientWidth || body.clientWidth
 71 				,scrollTop = docEl.scrollTop || body.scrollTop 
 72 				,scrollLeft = docEl.scrollLeft || body.scrollLeft 
 73 				,left = clientWidth * 0.35 + scrollLeft
 74 				,top = clientHeight * 0.25 + scrollTop;
 75 				
 76 			win.style.left = left + 'px';
 77        	 	win.style.top = top + 'px'; 
 78 		}
 79 		/**
 80 		 * 注销,取消拖拽
 81 		 */
 82 		,destory:function(titleDom) {
 83 			titleDom.style.cursor = 'default';
 84 			titleDom.onmousedown = null;
 85 		}
 86 		/**
 87 		 * 获取事件
 88 		 */
 89 		,_getEvent:function(){
 90 			 return window.event || arguments.callee.caller.arguments[0];
 91 		}
 92 	}
 93 })();
