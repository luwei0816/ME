1 /**
  2  * 文本框控件FDButton,继承自<a href="FDFieldComponent.html">FDFieldComponent</a><br>
  3  * @example 示例:<br>
  4  * <pre><code>
  5 var btn1;
  6 
  7 window.onload = function() {
  8 	btn1 = new FDButton({domId:'btn1',text:'按钮文本',onclick:function(){
  9 		alert(1)
 10 	}});
 11 }
 12 
 13 function setOnclickEvent() {
 14 	btn1.setOnclick(function(){
 15 		alert(2);
 16 	})
 17 }
 18 function setText() {
 19 	btn1.setText('按钮aaa')
 20 }</code></pre>
 21 * @param options 参见<a href="#getOptions">getOptions()</a><br>
 22 * @constructor
 23  */
 24 var FDButton = function(options) {
 25 	FDButton.superclass.constructor.call(this,options);
 26 	this.options = FDLib.util.apply(this.getOptions(),options);
 27 	
 28 	this.setOnclick(this.options.onclick);
 29 	this.setValue(this.options.text);
 30 	
 31 	this._initEvent();
 32 	
 33 	this.enable();
 34 }
 35 
 36 FDLib.extend(FDButton,FDFieldComponent);
 37 
 38 /**
 39  * 返回默认属性
 40  * @return <pre><code>
 41 {
 42 	// 按钮文本
 43 	text:''
 44 	// 事件函数
 45 	,onclick:null
 46 }</code></pre>
 47  */
 48 FDButton.prototype.getOptions = function() {
 49 	var options = FDButton.superclass.getOptions.call(this);
 50 	return FDLib.util.apply(options,{
 51 		text:''
 52 		,onclick:null
 53 	});
 54 }
 55 
 56 /**
 57  * 设置onclick事件
 58  * @param onclick 事件函数
 59  */
 60 FDButton.prototype.setOnclick = function(onclick) {
 61 	this.removeEvent('click',this.options.onclick);
 62 	this.addEvent('click',onclick);
 63 	this.options.onclick = onclick;
 64 }
 65 
 66 /**
 67  * 设置按钮文本
 68  * @param text 按钮文本
 69  */
 70 FDButton.prototype.setText= function(text) {
 71 	this.setValue(text);
 72 }
 73 
 74 /**
 75  * 禁用控件
 76  */
 77 // @override
 78 FDButton.prototype.enable = function() {
 79 	FDButton.superclass.enable.call(this);
 80 	this.setStyle('fdbutton fdbutton-enable');
 81 }
 82 
 83 /**
 84  * 启用控件
 85  */
 86 // @override
 87 FDButton.prototype.disable = function() {
 88 	FDButton.superclass.disable.call(this);
 89 	this.setStyle('fdbutton fdbutton-disable');
 90 }
 91 
 92 /**
 93  * @private
 94  */
 95 //@override
 96 FDButton.prototype.getControlHtml = function() {
 97 	return '<input type="button" />';
 98 }
 99 
100 FDButton.prototype._initEvent = function() {
101 	var self = this;
102 	
103 	this.addEvent('mouseover',function(){
104 		self.setStyle('fdbutton fdbutton-hover')
105 	});
106 	this.addEvent('mouseout',function(){
107 		self.setStyle('fdbutton fdbutton-enable')
108 	});
109 }
