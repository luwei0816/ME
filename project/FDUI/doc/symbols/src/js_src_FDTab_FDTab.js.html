1 /**
  2  * tab控件
  3  * @example 示例:
  4 var tab1;
  5 var tab2;
  6 window.onload = function() {
  7 	tab1 = new FDTab({domId:'tab1'
  8 		,isMouseover:true
  9 		,items:[
 10 		{text:'英文',value:1,contentId:'tab1-cont1',closeable:true,checked:false}
 11 		,{text:'中文',value:2,contentId:'tab1-cont2',checked:true}
 12 	]});
 13 	
 14 	
 15 	var en = 'After viewing the metamorphosis at sunrise, '
 16 		+ 'I would walk downhill along the steep mountain-path, '
 17 		+ 'towards the rocky beach, for a brief swim. Each time, '
 18 		+ 'I noticed a flurry of activity in a distant compound with a single decrepit building.';
 19 	
 20 	var zh = '在观看了日出时奇妙的变化以后，我会沿着陡峭的山路下行，走到一个遍布岩石的海滩，游一会儿泳。'
 21 	tab2 = new FDTab({domId:'tab2'
 22 		,items:[
 23 		{text:'英文2',value:3,content:'<i>'+en+'</i>',closeable:true,checked:false,onchange:changeHandler}
 24 		,{text:'中文2',value:4,content:'<b>'+zh+'</b>',closeable:true,checked:true,onchange:changeHandler}
 25 		,{text:"百度",closeable:true,url:'http://www.baidu.com',isRefresh:true,onclick:function(){alert('show baidu')},checked:false}
 26 	]});
 27 }
 28  * @param options 参见{@link #getOptions}返回的对象
 29  * @constructor
 30  */
 31 var FDTab = function(options) {
 32 	FDLib.implement(this,FDControl);
 33 	this.options = FDLib.util.apply(this.getOptions(),options);
 34 	// 根据value储存item
 35 	this.options.itemStore = {};
 36 	
 37 	this.tabInstance = new this.options.view(this.options,this);
 38 	
 39 	this._buildTab(this.options.items);
 40 }

 55 	// 	,onchange:function(item){}
 56 	// 	,checked:false

 58 	// }
 59 	,items:[]
 60 	//鼠标滑过切换tab,默认以click方式切换

 62 	,view:FDTabView

 64  */

233 	// 初始化items项
234 	,_initItems:function(items) {
235 		var self = this;
236 		FDLib.util.each(items,function(item){
237 			item = self._checkValueAndContentId(item);
238 			item = self._checkContentDiv(item);

249 	,_resetData:function() {
250 		this.options.itemStore = {};
251 		this.options.item = [];
252 	}
253 	,_delItemDataByValue:function(value) {
254 		var itemIndex = 0;
255 		var items = this.options.items;
256 		
257 		FDLib.util.each(items,function(item,i){
258 			if(item.value == value) {
259 				itemIndex = i;
260 				return false;
261 			}
262 		});
263 		
264 		this._delItemData(itemIndex);
265 		this._delItemStoreByValue(value);
266 	}
267 	,_delItemData:function(itemIndex) {
268 		var items = this.options.items;
269 		if(items.length > itemIndex){
270 		 	items.splice(itemIndex,1);
271 		}
272 	}
273 	,_delItemStoreByValue:function(value) {
274 		delete this.options.itemStore[value];
275 	}
276 	,_storeItem:function(item) {
277 		this.options.itemStore[item.value] = item;
278 	}
279 	// 确保value和contentId存在
280 	,_checkValueAndContentId:function(item) {
281 		var util = FDLib.util;
282 		var count = FDControl.generateCount();
283 		// 确保value存在
284 		if(util.isUndefined(item.value)) {
285 			item.value = 'tabValue_' + count;
286 		}
287 		if(util.isUndefined(item.contentId)) {
288 			item.contentId = 'tabContId_' + count;
289 		}
290 		return item;
291 	}
292 	// 确保content对应的div存在
293 	,_checkContentDiv:function(item) {
294 		// 保content存在
295 		if(FDLib.util.isUndefined(item.content)) {
296 			item.content = '';
297 		}
298 		var contentId = item.contentId,
299 			content = item.content;
300 		var contentDiv = null;
301 			
302 		if(contentId && FDLib.util.isString(contentId)) {
303 			contentDiv = FDLib.getEl(contentId);
304 		}
305 		// 如果contentDiv找不到,则创建一个空的div
306 		if(!contentDiv) {
307 			contentDiv = this._buildContentDiv(item);
308 		}
309 		item.contentDiv = contentDiv;
310 		
311 		return item;
312 	}
313 	,_buildContentDiv:function(item) {
314 		var contentId = item.contentId;
315 		var contentDiv = document.createElement(FDTag.DIV);
316 		contentDiv.setAttribute('id',contentId);
317 		contentDiv.innerHTML = item.content;
318 		
319 		this._buildIFrame(contentDiv,item);
320 		
321 		return contentDiv;
322 	}
323 	,_buildIFrame:function(contentDiv,item) {
324 		var url = item.url;
325 		if(FDLib.util.isString(url) && url) {
326 			var iframe = document.createElement(FDTag.IFRAME);
327 			//iframe.src = url; // 延迟加载
328 			iframe.style.cssText = "width:100%;height:100%;border:0px;";
329 			item.iframe = iframe;
330 			contentDiv.style.height = "100%";
331 			contentDiv.appendChild(iframe);
332 		}
333 	}
