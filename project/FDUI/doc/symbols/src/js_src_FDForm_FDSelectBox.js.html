1 /**
  2  * 选择框类FDSelectBox,继承自<a href="FDItemComponent.html">FDItemComponent</a><br>
  3  * @example 示例:
  4  * <pre></code>
  5 var select1;
  6 var select2;
  7 
  8 var genderItems = [{value:1,text:"男"},{value:0,text:"女"}];
  9 var constellationItems = [{value:0,text:"金牛座",date:'03-01'},{value:1,text:"天枰座",date:'04-01'}
 10 ,{value:2,text:"巨蟹座",date:'05-01'},{value:3,text:"双子座",date:'06-01'}];
 11 
 12 window.onload = function() {
 13 	select1 = new FDSelectBox({name:"gender",items:genderItems});
 14 	select1.renderTo('gender');
 15 	
 16 	select2 = new FDSelectBox({domId:'constellation',name:'constellation',items:constellationItems,label:'星坐:'});
 17 }</code></pre>
 18  * 2012-8-1
 19  * @param options 参见{@link #getOptions}返回的对象
 20  * @constructor
 21  */
 22 var FDSelectBox = function(options) {
 23 	FDSelectBox.superclass.constructor.call(this,options);
 24 }
 25 
 26 FDLib.extend(FDSelectBox,FDItemComponent);
 27 
 28 /**
 29  * 覆盖父类方法,获取组件的默认属性
 30    @return <pre><code>返回json数据类型
 31  * {
 32  *  domId:null // 定位的节点ID
 33 	,label:"" // 控件前面要显示的文字
 34 	,labelAlign:'right' // label水平对齐方式
 35 	,labelValign:'middle' // label垂直对齐方式
 36 	,name:"" // 控件的name
 37 	,width:"" // 控件的宽,如'120px'
 38 	,height:"" // 控件的高,如'30px'
 39 	,validates:[] // 验证类
 40 	,items:[] 
 41 	// 是否显示默认选项
 42 	,isShowDefaultItem:true
 43 	// 默认选项的文本
 44 	,defaultItemText:'-请选择-'
 45 	// 默认选项的值
 46 	,defaultItemValue:''
 47 	,defaultValue:''
 48  * }</code></pre>
 49  */
 50 //@override
 51 FDSelectBox.prototype.getOptions = function() {
 52 	var options = FDSelectBox.superclass.getOptions.call(this);
 53 	return FDLib.util.apply(options,{
 54 		// 是否显示默认选项
 55 		isShowDefaultItem:true
 56 		// 默认选项的文本
 57 		,defaultItemText:'-请选择-'
 58 		// 默认选项的值
 59 		,defaultItemValue:''
 60 		,defaultValue:''
 61 	});
 62 }
 63 
 64 /**
 65  * @private
 66  */
 67 // 重写父类的方法
 68 //@override
 69 FDSelectBox.prototype.buildItemsHtml = function() {
 70 	var name = this.options.name,
 71 		id = name + "_id";
 72 		
 73 	var selectStr = '<select id="'+id+'" name="'+name+'">'
 74 		+ this._getSelectItems() + '</select>';
 75 		
 76 	return selectStr;
 77 }
 78 
 79 FDSelectBox.prototype._getSelectItems = function() {
 80 	var defItemHtml = this._buildDefaultItem();
 81 	// 调用父类的buildItemsHtml方法
 82 	var itemsHtml = FDSelectBox.superclass.buildItemsHtml.call(this);
 83 	
 84 	return defItemHtml + itemsHtml;
 85 }
 86 
 87 FDSelectBox.prototype._buildDefaultItem = function() {
 88 	var defItem = '';
 89 	
 90 	if(this.options.isShowDefaultItem) {
 91 		var value = this.options.defaultItemValue;
 92 			text = this.options.defaultItemText
 93 		defItem = '<option value="'+value+'">'+text+'</option>';
 94 	}
 95 	return defItem;
 96 }
 97 
 98 /**
 99  * @private
100  */
101 //@override
102 FDSelectBox.prototype.getItemTemplate = function() {
103 	return '<option value="{value}">{text}</option>';
104 }
105

107  * @private
108  */
109 // 重写父类的方法
110 //@override
111 FDSelectBox.prototype.getControlItems = function() {
112 	return this.controlDiv.getElementsByTagName('select');
113 }
